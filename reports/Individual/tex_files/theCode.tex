\begin{lstlisting}{langauge=C}
for(j = 0; j < n; j++) {
    U[0][j] = A[0][j];
}

for(i = 0; i < n; i++) {
    L[i][i] = 1;
}

for(i = 1; i < n; i++) {
    L[i][0] = A[i][0]/U[0][0];
}

for(j = 1; j < n; j++) {
    for(i = 1; i <= j; i++) {
        for(k = 0; k <= i - 1; k++) {
            sum = sum + (L[i][k] * U[k][j]);
        }
        U[i][j] = A[i][j] - sum;
        sum = 0;
    }

    #pragma omp parallel for
    for(i = j + 1; i < n; i++) {
        for(k = 0; k <= j - 1; k++) {
            sum = sum + (L[i][k] * U[k][j]);
        }
        L[i][j] = (A[i][j] - sum) / U[j][j];
        sum = 0;
    }
}
\end{lstlisting}
